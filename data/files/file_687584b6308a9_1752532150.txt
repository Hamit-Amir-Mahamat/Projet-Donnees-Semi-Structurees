        function formatFileSize(bytes) {
            if (bytes === 0) return '0 Bytes';
            const k = 1024;
            const sizes = ['Bytes', 'KB', 'MB', 'GB'];
            const i = Math.floor(Math.log(bytes) / Math.log(k));
            return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
        }

        function formatTime(timestamp) {
            const now = new Date();
            const messageTime = new Date(timestamp);
            const diffTime = now - messageTime;
            const diffHours = Math.floor(diffTime / (1000 * 60 * 60));
            const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24));
            
            if (diffDays === 0) {
                return messageTime.toLocaleTimeString('fr-FR', { hour: '2-digit', minute: '2-digit' });
            } else if (diffDays === 1) {
                return 'Hier';
            } else if (diffDays < 7) {
                return messageTime.toLocaleDateString('fr-FR', { weekday: 'long' });
            } else {
                return messageTime.toLocaleDateString('fr-FR', { day: '2-digit', month: '2-digit' });
            }
        }
